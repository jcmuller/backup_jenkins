def parse_options
  options.each do |opt, arg|
    case opt
    when '--config'
      override_config_file_with(arg)
    when '--download'
      download_file(arg)
    when '--help'
      show_help
    when '--license'
      show_license
    when '--list'
      list_remote_backups
    when '--list-local'
      list_local_backups
    when '--only-local'
      @only_local = true
    when '--verbose'
      override_config_file_with_option("verbose" => true)
    when '--version'
      show_version
    end
  end
rescue GetoptLong::MissingArgument, GetoptLong::InvalidOption
  # GetoptLong Already outputs the error.
  puts "\n"
  show_help
end
