def do_backup
  raise "Backup directory already exists! (#{backup_directory})" if FileTest.directory?(backup_directory)

  copy_files
  create_tarball
  remove_temporary_files
  remove_old_backups
rescue Interrupt
  puts "Cleaning up..."
  clean_up
end
